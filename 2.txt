# 1. create folder
mkdir Git_Conflict_Demo

# 2. go inside
cd Git_Conflict_Demo

# 3. initialize git
git init

# 4. create file in main
echo "This is MAIN version" > file.txt

# 5. add & commit
git add file.txt
git commit -m "Initial commit in main"

# 6. create and switch branch
git checkout -b feature-branch

# 7. modify same file
echo "This is FEATURE branch change" > file.txt

# 8. add & commit
git add file.txt
git commit -m "Updated file in feature branch"

# 9. switch to main
git checkout main

# 10. modify same file differently
echo "This is MAIN branch change" > file.txt

# 11. add & commit
git add file.txt
git commit -m "Updated file in main branch"

# 12. merge feature branch
git merge feature-branch

# 13. open project in VS Code
code .

 In VS Code:

Click Accept Current Change (main) OR
Click Accept Incoming Change (feature) OR
Click Accept Both Changes

# 14. add resolved file
git add file.txt

# 15. commit merge
git commit -m "Resolved merge conflict"

# 16. check log
git log --oneline

# 17. check final content
type file.txt
